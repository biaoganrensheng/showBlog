1./App.js

import React, { Component } from 'react';
import logo from './logo.svg';
import {Provider} from 'react-redux'

import PostMsg from './components/Post'
import PostForm from './components/PostForm'
import './App.css';

import {store} from './store'
class App extends Component {
  render() {
    return (
      <Provider store={store}>
      <div className="App">
        <header className="App-header">
          <img src={logo} className="App-logo" alt="logo" />
          <h1 className="App-title">Welcome to React</h1>
        </header>
        <PostForm/>
        <PostMsg/>
      </div>
      </Provider>
    );
  }
}

export default App;

2./store.js

import {createStore,applyMiddleware,compose} from 'redux'
import thunk from 'redux-thunk'
import rootReducer from './reducer'
const initState={};
const middleware=[thunk]

let store;
if(!(window.__REDUX_DEVTOOLS_EXTENSION__ || window.__REDUX_DEVTOOLS_EXTENSION__)){
    store = createStore(
       rootReducer,
        applyMiddleware(...middleware)
    );
}else{
    store =createStore(rootReducer,{},compose(applyMiddleware(...middleware),window.__REDUX_DEVTOOLS_EXTENSION__ && window.__REDUX_DEVTOOLS_EXTENSION__()))
}
export  store

4.reducer/postreducer.js

import {FETCH_POSTS,FETCH_POST_USER} from '../actions/types'
const initState={
    items:[],
    user:{}
};

export default function(state=initState,action){
    switch(action.type){
        case FETCH_POSTS:
        return {
            ...state,
            items:action.paylod
        }
        break;
        case FETCH_POST_USER:
        return {
            ...state,
            user:action.paylod
        }
        break;
        default:
        return state;
    }
}

import React, { Component } from 'react'
import {fetchPost} from '../actions/postAction'
import PropTypes from 'prop-types'
import {connect} from 'react-redux'
class PostMsg extends Component {
    // constructor(){
    //     super();
    //     this.state={
    //         info:[]
    //     }
    // }
    componentDidMount(){
        this.props.fetchPost();
        // fetch("http://jsonplaceholder.typicode.com/posts")
        //     .then(res=>res.json())
        //     .then(res=>this.setState({
        //         info:res
        //     }))
    }
    componentWillReceiveProps(nextprop){
        if(nextprop.user){
            this.props.post.unshift(nextprop.user)
        }    
    }
    render() {
        const info=this.props.post.map((post,index)=>{
            return (
            <div key={post.id}>
                <h3>{post.title}</h3>
                <p>{post.body}</p>    
            </div>
            )
        })
        return (
            <div>
            {info}
            </div>
        )
    }
}
PostMsg.propTypes={
    fetchPost:PropTypes.func.isRequired,
    post:PropTypes.array.isRequired
}
const mapStateToProps = (state) => ({
   post:state.post.items,
   user:state.post.user
  }) 
export default connect(mapStateToProps,{fetchPost})(PostMsg);
